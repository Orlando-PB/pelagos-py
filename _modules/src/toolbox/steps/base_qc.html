def register_qc(cls):
    """Decorator to mark QC tests that can be accessed by the ApplyQC step."""
    qc_name = getattr(cls, "qc_name", None)
    if qc_name is None:
        raise ValueError(
            f"QC test {cls.__name__} is missing required 'qc_name' attribute."
        )
    REGISTERED_QC[qc_name] = cls
    return cls
